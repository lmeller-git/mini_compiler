begin_def main;
	print_str "Compute wether number is prime\n";

	number = 9999996000003199;

	print_str "the number generated by typing randomly is: ";
	print number;
	print_str "\n";

	# compute upper bound for loop
	upper_bound = number;
	sqrt &upper_bound;

	print_str "the upper bound is: ";
	print upper_bound;
	print_str "\n";

	check = 2;
	label loop;
	if !(number % check); goto not_prime;

	check = check + 1;

	if check > upper_bound; goto prime;

	goto loop;


	label not_prime;
	print_str "the number is not a prime\n";
	goto exit_;


	label prime;
	print_str "the number is a prime\n";


	label exit_;
end_def
